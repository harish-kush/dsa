class Solution {
public:
    void helper(TreeNode* root, vector<int>& ans){
        if(root==NULL) return ;
        helper(root->left,ans);
        ans.push_back(root->val);
        helper(root->right, ans);
    }
    int kthSmallest(TreeNode* root, int k) {
        vector<int>ans;
        helper(root,ans);
        
        return ans[k-1];
    }
};
