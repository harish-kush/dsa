class Solution {
public:
    ListNode* mergeInBetween(ListNode* list1, int a, int b, ListNode* list2) {
        ListNode* ta = list1;
        ListNode* tb = list1;
        ListNode* tc = list2;
        for(int i=0;i<a-1;i++){
            ta = ta->next;
        } 
        for(int i=0;i<b+1;i++){
            tb = tb->next;
        } 
        ta->next = tc;
        while(tc->next){
            tc = tc->next;
        }
        tc->next = tb;
        return list1;
    }
};
