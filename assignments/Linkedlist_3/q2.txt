class Solution {
public:
    ListNode* insertionSortList(ListNode* head) {
        ListNode* temp = head;
        vector<int>v;
        int n=0;
        while(temp){
            v.push_back(temp->val);
            n++;
            temp = temp->next;
        }
        temp = head;
        sort(v.begin(),v.end());
        int i=0;
        while(temp){
            temp->val = v[i];
            temp = temp->next;
            i++;
        }
        return head;
    }
};
