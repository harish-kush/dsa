class Solution {
public:
    int getDecimalValue(ListNode* head) {
   int n=0;
   ListNode* temp = head;
   while(temp){
    n++;
    temp = temp->next;
   }     
   temp = head;
   int sum=0;
   while(temp){
    sum+= temp->val * pow(2,n-1);
    n--;
    temp = temp->next;
   }
   return sum;
    }
};
