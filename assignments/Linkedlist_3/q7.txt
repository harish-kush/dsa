class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
    if(head==NULL) return NULL;
    ListNode* prev = NULL;
    ListNode* next = NULL;
    ListNode* curr = head;
    int c = k;
    while(curr!=NULL && c>0){
        next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
        c--;
    }
    if(curr==NULL && c>0){
        curr = prev;
        prev = NULL;
        next = NULL;
        while(curr){
        next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
        }
    }
    if(c>0) return prev;
    head->next=reverseKGroup(curr,k);
    return prev;
    }
};
